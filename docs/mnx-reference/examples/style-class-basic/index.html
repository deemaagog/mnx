Styling via a class (basic)

This document defines a style class called "emphasized". Any element with class="emphasized" will be colored blue.

{ "global": { "measures": [ { "barline": { "type": "regular" }, "time": { "signature": "4/4" } } ], "styles": [ { "color": "#0080FF", "selector": ".emphasized" } ] }, "parts": [ { "measures": [ { "content": [ { "type": "clef", "line": 2, "sign": "G" }, { "type": "sequence", "content": [ { "type": "event", "notes": [ { "class": "emphasized", "pitch": "C5" } ], "value": "/4" }, { "type": "event", "notes": [ { "pitch": "E5" } ], "value": "/4" }, { "type": "event", "notes": [ { "pitch": "G5" } ], "value": "/4" }, { "type": "event", "notes": [ { "class": "emphasized", "pitch": "C6" } ], "value": "/4" } ] } ] } ] } ] }
